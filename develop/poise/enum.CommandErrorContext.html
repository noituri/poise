async fn on_error(error: Error, ctx: poise::ErrorContext<'_, Data, Error>) {
    match ctx {
        poise::ErrorContext::Command(ctx) => {
            println!(
                "Command '{}' returned error {:?}",
                ctx.command().name(),
                error
            );
        }
        poise::ErrorContext::Listener(event) => {
            println!(
                "Listener returned error during {} event: {:?}",
                event.name(),
                error
            );
        }
        _ => {}
    }
}
